\section{Problema 1 -- Proceso de inyecci\'on de pl\'astico (grosor del envase)}

\noindent\textbf{Datos del problema:} Grosor nominal = 1.20\,mm,
tolerancia $\pm\,0.10$\,mm, muestra $n = 125$ piezas.
\[
  E_I = 1.10\;\text{mm} \qquad E_S = 1.30\;\text{mm}
\]

% ─── 1.A Tendencia central ────────────────────────────────────────────────────
\subsection{A) Medidas de tendencia central}

\subsubsection*{Media aritm\'etica ($\bar{x}$)}

La media se obtiene sumando todos los valores y dividiendo entre $n$:
\[
\bar{x} = \frac{\displaystyle\sum_{i=1}^{n} x_i}{n}
        = \frac{1.15 + 1.20 + 1.17 + \cdots + 1.18}{125}
        = \frac{147.33}{125}
\]

\resultado[$\bar{x}$ -- Media]{\[ \bar{x} = 1.1787\;\text{mm} \]}

\subsubsection*{Mediana ($\tilde{x}$)}

Con $n = 125$ (impar), la posici\'on central es:
\[
\text{posici\'on} = \frac{n+1}{2} = \frac{126}{2} = 63
\]
Al ordenar los 125 datos de menor a mayor, el valor en la posici\'on~63 es
$1.18\,\text{mm}$.

\resultado[$\tilde{x}$ -- Mediana]{\[ \tilde{x} = 1.18\;\text{mm} \]}

\subsubsection*{Moda ($M_o$)}

Contando la frecuencia de cada valor \'unico, los cuatro m\'as repetidos son:

\begin{center}
\begin{tabular}{cc}
\toprule
\rowcolor{azulclaro}
\textbf{Valor (mm)} & \textbf{Frecuencia} \\
\midrule
1.17 & \textbf{23} \\
\rowcolor{grisclaro} 1.19 & 19 \\
1.16 & 16 \\
\rowcolor{grisclaro} 1.20 & 15 \\
\bottomrule
\end{tabular}
\end{center}

\resultado[$M_o$ -- Moda]{\[ M_o = 1.17\;\text{mm} \quad \text{(aparece 23 veces)} \]}

\interpretacion[Tendencia central --- Problema 1]{
La media (1.1787\,mm), la mediana (1.18\,mm) y la moda (1.17\,mm) son muy
pr\'oximas entre s\'i, lo que indica una distribuci\'on \textbf{aproximadamente
sim\'etrica}. La media se encuentra \textbf{dentro} del rango permitido
$[1.10,\,1.30]$\,mm, aunque ligeramente por debajo del nominal de 1.20\,mm.
La tendencia central del proceso es \textbf{adecuada}.}

% ─── 1.B Dispersión ───────────────────────────────────────────────────────────
\subsection{B) Medidas de dispersi\'on}

\subsubsection*{Rango ($R$)}

\[
R = x_{\max} - x_{\min} = 1.25 - 1.11 = 0.14\;\text{mm}
\]

\resultado[$R$ -- Rango]{\[ R = 0.14\;\text{mm} \]}

\subsubsection*{Varianza muestral ($s^2$)}

La varianza muestral mide el promedio de los cuadrados de las desviaciones
respecto a la media:
\[
s^2 = \frac{\displaystyle\sum_{i=1}^{n}(x_i - \bar{x})^2}{n-1}
\]
Con $\bar{x} = 1.1787$\,mm, algunos t\'erminos representativos son:
\begin{align*}
(1.15 - 1.1787)^2 &= (-0.0287)^2 = 0.000824 \\
(1.20 - 1.1787)^2 &= (\phantom{-}0.0213)^2 = 0.000454 \\
(1.11 - 1.1787)^2 &= (-0.0687)^2 = 0.004720 \\[-2pt]
                   &\hspace{6cm}\vdots
\end{align*}
La suma de todos los cuadrados de desviaci\'on resulta $0.090288$\,mm$^2$.
Dividiendo entre $n-1 = 124$:
\[
s^2 = \frac{0.090288}{124} = 0.000728\;\text{mm}^2
\]

\resultado[$s^2$ -- Varianza]{\[ s^2 = 0.000728\;\text{mm}^2 \]}

\subsubsection*{Desviaci\'on est\'andar muestral ($s$)}

\[
s = \sqrt{s^2} = \sqrt{0.000728} = 0.02698\;\text{mm}
\]

\resultado[$s$ -- Desviaci\'on est\'andar]{\[ s = 0.02698\;\text{mm} \approx 0.027\;\text{mm} \]}

\subsubsection*{Coeficiente de variaci\'on ($CV$)}

El coeficiente de variaci\'on expresa la desviaci\'on est\'andar como porcentaje
de la media:
\[
CV = \frac{s}{\bar{x}} \times 100\%
   = \frac{0.02698}{1.1787} \times 100\% = 2.29\%
\]

\resultado[$CV$ -- Coeficiente de variaci\'on]{\[ CV = 2.29\% \]}

\interpretacion[Dispersi\'on --- Problema 1]{
El $CV = 2.29\% < 10\%$ indica una variabilidad \textbf{muy baja y aceptable}.
El rango de 0.14\,mm est\'a contenido dentro de la tolerancia total de 0.20\,mm.
El proceso tiene \textbf{poca variabilidad}.}

\section{Problema 2 -- Corte de tiras de hule (longitud)}

\noindent\textbf{Datos del problema:} Longitud nominal = 200\,mm,
tolerancia $\pm\,3$\,mm, muestra $n = 110$ piezas (dos m\'aquinas).
\[
  E_I = 197\;\text{mm} \qquad E_S = 203\;\text{mm}
\]

% ─── 2.A Tendencia central ────────────────────────────────────────────────────
\subsection{A) Medidas de tendencia central}

\subsubsection*{Media aritm\'etica ($\bar{x}$)}

Sumando los 110 valores y dividiendo entre $n = 110$:
\[
\bar{x} = \frac{\displaystyle\sum_{i=1}^{n} x_i}{n}
        = \frac{199.2 + 199.7 + 201.8 + \cdots + 198.5}{110}
        = \frac{21\,999.60}{110}
\]

\resultado[$\bar{x}$ -- Media]{\[ \bar{x} = 199.9964\;\text{mm} \approx 200.00\;\text{mm} \]}

\subsubsection*{Mediana ($\tilde{x}$)}

Con $n = 110$ (par), las dos posiciones centrales son $n/2 = 55$ y $n/2 + 1 = 56$.
Al ordenar los datos, los valores en esas posiciones son
$x_{(55)} = 200.0\;\text{mm}$ y $x_{(56)} = 200.1\;\text{mm}$, por lo que:
\[
\tilde{x} = \frac{200.0 + 200.1}{2} = 200.05\;\text{mm}
\]

\resultado[$\tilde{x}$ -- Mediana]{\[ \tilde{x} = 200.05\;\text{mm} \]}

\subsubsection*{Moda ($M_o$)}

Los tres valores con mayor frecuencia son:

\begin{center}
\begin{tabular}{cc}
\toprule
\rowcolor{azulclaro}
\textbf{Valor (mm)} & \textbf{Frecuencia} \\
\midrule
199.0 & \textbf{6} \\
\rowcolor{grisclaro} 200.5 & \textbf{6} \\
201.4 & \textbf{6} \\
\bottomrule
\end{tabular}
\end{center}

\resultado[$M_o$ -- Moda]{
  Distribuci\'on \textbf{trimodal}:
  $M_o = 199.0,\; 200.5,\; 201.4$\,mm \quad ($f = 6$ cada una)}

\interpretacion[Tendencia central --- Problema 2]{
La media ($\approx 200.00$\,mm) est\'a \textbf{pr\'acticamente sobre el valor nominal}
de 200\,mm. La mediana (200.05\,mm) tambi\'en es muy cercana al nominal.
La distribuci\'on trimodal indica que las \textbf{dos m\'aquinas}
producen longitudes con centros ligeramente distintos. En general, la
tendencia central del proceso es \textbf{adecuada}.}

% ─── 2.C Dispersión ───────────────────────────────────────────────────────────
\subsection{C) Medidas de dispersi\'on}

\subsubsection*{Rango ($R$)}

\[
R = x_{\max} - x_{\min} = 202.1 - 197.8 = 4.30\;\text{mm}
\]

\resultado[$R$ -- Rango]{\[ R = 4.30\;\text{mm} \]}

\subsubsection*{Varianza muestral ($s^2$)}

\[
s^2 = \frac{\displaystyle\sum_{i=1}^{n}(x_i - \bar{x})^2}{n-1}
\]
Con $\bar{x} = 199.9964$\,mm, algunos t\'erminos representativos son:
\begin{align*}
(199.2 - 199.9964)^2 &= (-0.7964)^2 = 0.6342 \\
(202.1 - 199.9964)^2 &= (\phantom{-}2.1036)^2 = 4.4251 \\
(197.8 - 199.9964)^2 &= (-2.1964)^2 = 4.8242 \\[-2pt]
                      &\hspace{6cm}\vdots
\end{align*}
La suma total de todos los cuadrados de desviaci\'on es $146.914$\,mm$^2$.
Dividiendo entre $n - 1 = 109$:
\[
s^2 = \frac{146.914}{109} = 1.3478\;\text{mm}^2
\]

\resultado[$s^2$ -- Varianza]{\[ s^2 = 1.3478\;\text{mm}^2 \]}

\subsubsection*{Desviaci\'on est\'andar ($s$)}

\[
s = \sqrt{s^2} = \sqrt{1.3478} = 1.1610\;\text{mm}
\]

\resultado[$s$ -- Desviaci\'on est\'andar]{\[ s = 1.1610\;\text{mm} \]}

\subsubsection*{Coeficiente de variaci\'on ($CV$)}

\[
CV = \frac{s}{\bar{x}} \times 100\%
   = \frac{1.1610}{199.9964} \times 100\% = 0.581\%
\]

\resultado[$CV$ -- Coeficiente de variaci\'on]{\[ CV = 0.581\% \]}

\interpretacion[Dispersi\'on --- Problema 2]{
El $CV = 0.581\% \ll 10\%$ indica una variabilidad \textbf{extremadamente baja
y aceptable}. La desviaci\'on est\'andar de 1.16\,mm representa solo el 19\%
de la franja de tolerancia total (6\,mm). La variabilidad del proceso es
\textbf{aceptable}.}

\section{Pregunta 3 -- Caracter\'istica de calidad del proyecto de ingenier\'ia}

\noindent\textbf{Proyecto:} \textit{PharmaTrack} --- Sistema de monitoreo de condiciones
de almacenamiento de medicamentos mediante sensores IoT
(ICE, IPN ESIME Culhuac\'an).

\vspace{8pt}

\begin{tcolorbox}[resultadoStyle, title={Caracter\'istica de calidad seleccionada}]
\textbf{Temperatura de almacenamiento} de los medicamentos monitoreada por los
nodos sensores del sistema PharmaTrack.

\vspace{6pt}
\begin{tabular}{@{}ll@{}}
  \textbf{Variable:}    & Temperatura ($T$) \\[3pt]
  \textbf{Unidad:}      & Grados Celsius (\si{\celsius}) \\[3pt]
  \textbf{Instrumento:} & Sensor digital DS18B20
                          ($\pm$\,0.5\,\si{\celsius} de resoluci\'on) \\[3pt]
  \textbf{Nominal:}     & $T_{\text{nom}} = 5$\,\si{\celsius}
                          (cadena de fr\'io farmac\'eutica) \\[3pt]
  \textbf{Tolerancia:}  & $\pm\,3$\,\si{\celsius},\quad
                          $T \in [2,\,8]$\,\si{\celsius}
                          seg\'un NOM-059-SSA1
\end{tabular}
\end{tcolorbox}

\interpretacion[Justificaci\'on]{
La temperatura es la caracter\'istica de calidad cr\'itica en PharmaTrack porque
una desviaci\'on fuera del rango $[2,\,8]$\,\si{\celsius} degrada la eficacia o
la seguridad de los medicamentos. El sensor DS18B20 permite adquisici\'on continua
v\'ia protocolo 1-Wire; su se\~nal es procesada por la API del sistema y visualizada
en el \textit{dashboard} del \textit{frontend}, lo que habilita alertas en tiempo real
y trazabilidad completa de la cadena de fr\'io.}

\color{white}\textbf{Qu\'e eval\'ua / mide} \\
\midrule
4 & Contexto de la organizaci\'on &
Factores internos y externos, partes interesadas, alcance del SGC. \\
\rowcolor{grisclaro}
5 & Liderazgo &
Compromiso de la alta direcci\'on, pol\'itica de calidad, roles y responsabilidades. \\
6 & Planificaci\'on &
Riesgos y oportunidades, objetivos de calidad, planificaci\'on de cambios. \\
\rowcolor{grisclaro}
7 & Soporte &
Recursos (humanos, infraestructura, ambiente), competencias, comunicaci\'on,
informaci\'on documentada. \\
8 & Operaci\'on &
Control de procesos, dise\~no, producci\'on, provisi\'on de servicios y
control de salidas no conformes. \\
\rowcolor{grisclaro}
9 & Evaluaci\'on del desempe\~no &
Seguimiento, medici\'on y an\'alisis; auditor\'ias internas; revisi\'on por la
direcci\'on; satisfacci\'on del cliente. \\
10 & Mejora &
No conformidades, acciones correctivas y mejora continua. \\
\bottomrule
\end{tabular}
\end{center}

\interpretacion[Conclusi\'on]{
La norma no eval\'ua solo el producto final: mide el \textbf{sistema completo}
que rodea su producci\'on. Para PharmaTrack, esto implica demostrar conformidad
en todas las cl\'ausulas --- desde el contexto organizacional hasta las acciones
correctivas --- para garantizar que el software, el hardware y los procesos
operativos produzcan resultados de calidad de forma consistente y trazable.}

\section{Pregunta 5 -- Ciclo PHVA en ISO\,9001:2015 y NMX-CC-9001}

\noindent El \textbf{Ciclo PHVA} (Planear -- Hacer -- Verificar -- Actuar),
tambi\'en conocido como \textit{ciclo de Deming} o \textit{PDCA} en ingl\'es,
es el modelo iterativo de mejora continua sobre el que se estructura la norma
ISO\,9001:2015. Cada etapa corresponde a un conjunto de cl\'ausulas:

\vspace{8pt}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{>{\centering\bfseries\color{azulprin}}m{1.8cm}
                >{\centering}m{2.2cm}
                >{\raggedright\arraybackslash}m{8.5cm}}
\toprule
\rowcolor{azulprin}
\color{white}\textbf{Etapa} &
\color{white}\textbf{Cl\'ausulas} &
\color{white}\textbf{Descripci\'on} \\
\midrule
Planear & 4, 5, 6, 7 &
Establecer objetivos, identificar riesgos y oportunidades, asignar recursos
y definir los procesos necesarios para obtener resultados conformes con los
requisitos del cliente. \\
\rowcolor{grisclaro}
Hacer & 8 &
Implementar y ejecutar los procesos planificados: producci\'on, prestaci\'on
del servicio, control de proveedores y gesti\'on de cambios. \\
Verificar & 9 &
Hacer seguimiento y medici\'on de los procesos y resultados; realizar
auditor\'ias internas; analizar satisfacci\'on del cliente y reportar
el desempe\~no a la direcci\'on. \\
\rowcolor{grisclaro}
Actuar & 10 &
Tomar acciones correctivas ante no conformidades e implementar mejora
continua; los resultados realimentan la etapa de Planear. \\
\bottomrule
\end{tabular}
\end{center}

\vspace{10pt}
\begin{tcolorbox}[enhanced, colback=azulclaro, colframe=azulborde, arc=4pt,
    boxrule=1pt, left=8pt, right=8pt, top=6pt, bottom=6pt]
\centering
\textbf{\color{azulprin}El ciclo es iterativo e infinito:}\\[4pt]
$\text{Planear} \;\longrightarrow\; \text{Hacer} \;\longrightarrow\;
 \text{Verificar} \;\longrightarrow\; \text{Actuar} \;\longrightarrow\;
 \text{Planear} \;\longrightarrow\; \cdots$\\[4pt]
Cada vuelta completa eleva el nivel de calidad del sistema.
\end{tcolorbox}

\interpretacion[Aplicaci\'on en PharmaTrack]{
\textbf{Planear:} definir el rango aceptable $[2,\,8]$\,\si{\celsius} y los umbrales
de alerta del sistema.
\textbf{Hacer:} operar los nodos sensores y registrar las lecturas en la API.
\textbf{Verificar:} analizar los datos en el \textit{dashboard}; detectar desviaciones
y generar reportes.
\textbf{Actuar:} emitir alertas, corregir fallas en la cadena de fr\'io y ajustar los
par\'ametros para la siguiente iteraci\'on.}

\section{Pregunta 7 -- Trazabilidad de las mediciones}

\begin{tcolorbox}[resultadoStyle, title={Definici\'on}]
La \textbf{trazabilidad de las mediciones} es la propiedad de un resultado de
medici\'on por la cual ese resultado puede relacionarse con una referencia
nacional o internacional mediante una \textbf{cadena ininterrumpida y documentada
de calibraciones}, cada una con su incertidumbre asociada
(ISO\,9001:2015, cl\'ausula~7.1.5).
\end{tcolorbox}

\vspace{10pt}
\noindent\textbf{Cadena de trazabilidad --- temperatura en PharmaTrack:}

\vspace{6pt}
\begin{center}
\renewcommand{\arraystretch}{1.6}
\begin{tabular}{>{\centering}m{0.5cm}
                >{\bfseries\centering}m{3.8cm}
                >{\raggedright\arraybackslash}m{8cm}}
\toprule
\rowcolor{azulprin}
\color{white}\textbf{\#} &
\color{white}\textbf{Nivel} &
\color{white}\textbf{Descripci\'on} \\
\midrule
1 & Patr\'on internacional &
Sistema Internacional de Unidades (SI) definido por el BIPM
(\textit{Bureau International des Poids et Mesures}). \\
\rowcolor{grisclaro}
2 & Patr\'on nacional &
CENAM (Centro Nacional de Metrolog\'ia, M\'exico) --- custodio del patr\'on
primario de temperatura en M\'exico. \\
3 & Laboratorio acreditado &
Laboratorio con acreditaci\'on EMA que calibra term\'ometros de referencia
trazables al CENAM. \\
\rowcolor{grisclaro}
4 & Instrumento de campo &
Sensor DS18B20 calibrado contra el term\'ometro de referencia del laboratorio
acreditado, con certificado y fecha de vencimiento. \\
\bottomrule
\end{tabular}
\end{center}

\interpretacion[Importancia de la trazabilidad]{
Sin trazabilidad, una lectura de $5$\,\si{\celsius} del sensor PharmaTrack no tiene
significado metrolog\'ico verificable. Con trazabilidad, cada dato registrado
puede defenderse documentalmente ante una auditor\'ia, cumpliendo la cl\'ausula~7.1.5
de ISO\,9001:2015 y los requisitos de la NOM-059-SSA1 para la cadena de fr\'io
farmac\'eutica.}
