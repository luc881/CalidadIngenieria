\section{Problema 1 -- Proceso de inyecci\'on de pl\'astico (grosor del envase)}

\noindent\textbf{Datos del problema:} Grosor nominal = 1.20\,mm, tolerancia $\pm$0.10\,mm.
\[
  E_I = 1.10\;\text{mm} \qquad E_S = 1.30\;\text{mm} \qquad n = 125
\]

%% -------------------------------------------------------------------
\subsection{Medidas de tendencia central}

\subsubsection*{Media aritm\'etica ($\bar{x}$)}

\paso{1} Escribir la f\'ormula de la media:
\formula{\bar{x} = \frac{\displaystyle\sum_{i=1}^{n} x_i}{n}}

\paso{2} Sumar los 125 valores del enunciado:
\formula{\sum_{i=1}^{125} x_i = 1.15 + 1.20 + 1.17 + \cdots + 1.18 = 147.33}

\paso{3} Dividir entre $n = 125$:
\formula{\bar{x} = \frac{147.33}{125} = 1.1787\;\text{mm}}

\resultado[$\bar{x}$ -- Media]{\[ \bar{x} = 1.1787\;\text{mm} \]}

% ------
\subsubsection*{Mediana ($\tilde{x}$)}

\paso{1} Con $n = 125$ (impar), calcular la posici\'on central:
\formula{\text{posici\'on} = \frac{n+1}{2} = \frac{126}{2} = 63}

\paso{2} Ordenar los 125 datos de menor a mayor. El valor en la posici\'on 63 es:

\resultado[$\tilde{x}$ -- Mediana]{\[ \tilde{x} = 1.18\;\text{mm} \]}

% ------
\subsubsection*{Moda ($M_o$)}

\paso{1} Contar la frecuencia de cada valor \'unico. Los m\'as frecuentes son:

\begin{center}
\begin{tabular}{cc}
\toprule
\rowcolor{azulclaro}
\textbf{Valor (mm)} & \textbf{Frecuencia} \\
\midrule
1.17 & \textbf{23} \\
\rowcolor{grisclaro} 1.19 & 19 \\
1.16 & 16 \\
\rowcolor{grisclaro} 1.20 & 15 \\
\bottomrule
\end{tabular}
\end{center}

\resultado[$M_o$ -- Moda]{\[ M_o = 1.17\;\text{mm} \quad \text{(aparece 23 veces)} \]}

\interpretacion[Tendencia central --- Problema 1]{
La media (1.1787\,mm), la mediana (1.18\,mm) y la moda (1.17\,mm) son muy
pr\'oximas entre s\'i, lo que indica una distribuci\'on \textbf{aproximadamente
sim\'etrica}. La media se encuentra \textbf{dentro} del rango permitido
[1.10,\,1.30]\,mm, aunque ligeramente por debajo del nominal de 1.20\,mm.
La tendencia central del proceso es \textbf{adecuada}.}

%% -------------------------------------------------------------------
\subsection{Medidas de dispersi\'on}

\subsubsection*{Rango ($R$)}

\paso{1} Identificar m\'inimo y m\'aximo:
\formula{x_{\min} = 1.11\;\text{mm}, \qquad x_{\max} = 1.25\;\text{mm}}

\paso{2} Calcular:
\formula{R = x_{\max} - x_{\min} = 1.25 - 1.11 = 0.14\;\text{mm}}

\resultado[$R$ -- Rango]{\[ R = 0.14\;\text{mm} \]}

% ------
\subsubsection*{Varianza muestral ($s^2$)}

\paso{1} F\'ormula de la varianza muestral:
\formula{s^2 = \frac{\displaystyle\sum_{i=1}^{n}(x_i - \bar{x})^2}{n-1}}

\paso{2} Con $\bar{x} = 1.1787$\,mm, calcular tres t\'erminos representativos:
\begin{align*}
(1.15 - 1.1787)^2 &= (-0.0287)^2 = 0.000824 \\
(1.20 - 1.1787)^2 &= (\phantom{-}0.0213)^2 = 0.000454 \\
(1.11 - 1.1787)^2 &= (-0.0687)^2 = 0.004720 \\
&\vdots
\end{align*}

\paso{3} Sumar todos los cuadrados de desviaci\'on:
\formula{\sum_{i=1}^{125}(x_i - \bar{x})^2 = 0.090288}

\paso{4} Dividir entre $n - 1 = 124$:
\formula{s^2 = \frac{0.090288}{124} = 0.000728\;\text{mm}^2}

\resultado[$s^2$ -- Varianza]{\[ s^2 = 0.000728\;\text{mm}^2 \]}

% ------
\subsubsection*{Desviaci\'on est\'andar muestral ($s$)}

\paso{1} F\'ormula:
\formula{s = \sqrt{s^2}}

\paso{2} Sustituir:
\formula{s = \sqrt{0.000728} = 0.02698\;\text{mm}}

\resultado[$s$ -- Desviaci\'on est\'andar]{\[ s = 0.02698\;\text{mm} \approx 0.027\;\text{mm} \]}

% ------
\subsubsection*{Coeficiente de variaci\'on ($CV$)}

\paso{1} F\'ormula:
\formula{CV = \frac{s}{\bar{x}} \times 100\%}

\paso{2} Sustituir con $s = 0.02698$ y $\bar{x} = 1.1787$:
\formula{CV = \frac{0.02698}{1.1787} \times 100\% = 2.29\%}

\resultado[$CV$ -- Coeficiente de variaci\'on]{\[ CV = 2.29\% \]}

\interpretacion[Dispersi\'on --- Problema 1]{
El $CV = 2.29\% < 10\%$ indica una variabilidad \textbf{muy baja y aceptable}.
El rango de 0.14\,mm est\'a contenido dentro de la tolerancia total de 0.20\,mm.
El proceso tiene \textbf{poca variabilidad}.}

\section{Problema 2 -- Corte de tiras de hule (longitud)}

\noindent\textbf{Datos del problema:} Longitud nominal = 200\,mm, tolerancia $\pm$3\,mm.
\[
  E_I = 197\;\text{mm} \qquad E_S = 203\;\text{mm} \qquad n = 110
\]

%% -------------------------------------------------------------------
\subsection{A) Medidas de tendencia central}

\subsubsection*{Media aritm\'etica ($\bar{x}$)}

\paso{1} F\'ormula:
\formula{\bar{x} = \frac{\displaystyle\sum_{i=1}^{n} x_i}{n}}

\paso{2} Sumar los 110 valores:
\formula{\sum_{i=1}^{110} x_i = 199.2 + 199.7 + 201.8 + \cdots + 198.5 = 21{,}999.60}

\paso{3} Dividir entre $n = 110$:
\formula{\bar{x} = \frac{21{,}999.60}{110} = 199.9964\;\text{mm}}

\resultado[$\bar{x}$ -- Media]{\[ \bar{x} = 199.9964\;\text{mm} \approx 200.00\;\text{mm} \]}

% ------
\subsubsection*{Mediana ($\tilde{x}$)}

\paso{1} Con $n = 110$ (par), las posiciones centrales son:
\formula{\frac{n}{2} = 55 \qquad \frac{n}{2}+1 = 56}

\paso{2} Ordenar los datos. Los valores en esas posiciones son:
$x_{(55)} = 200.0\;\text{mm}$\quad y\quad $x_{(56)} = 200.1\;\text{mm}$.

\paso{3} Calcular:
\formula{\tilde{x} = \frac{200.0 + 200.1}{2} = 200.05\;\text{mm}}

\resultado[$\tilde{x}$ -- Mediana]{\[ \tilde{x} = 200.05\;\text{mm} \]}

% ------
\subsubsection*{Moda ($M_o$)}

\paso{1} Contar frecuencias. Los valores con mayor repetici\'on son:

201.4 & \textbf{6} \\
\bottomrule

\end{center}

\resultado[$M_o$ -- Moda]{
  Distribuci\'on \textbf{trimodal}: $M_o = 199.0,\; 200.5,\; 201.4$ mm\quad ($f = 6$ cada una)}

\interpretacion[Tendencia central --- Problema 2]{
La media ($\approx 200.00$\,mm) est\'a \textbf{pr\'acticamente sobre el valor nominal}
de 200\,mm. La mediana (200.05\,mm) tambi\'en es muy cercana al nominal.
La distribu\-ci\'on trimodal puede indicar que las \textbf{dos m\'aquinas}
producen longitudes con centros ligeramente distintos. En general, la
tendencia central del proceso es \textbf{adecuada}.}

\subsection{C) Desviaci\'on est\'andar y coeficiente de variaci\'on}

\subsubsection*{Rango ($R$)}

\formula{R = x_{\max} - x_{\min} = 202.1 - 197.8 = 4.30\;\text{mm}}

\resultado[$R$ -- Rango]{\[ R = 4.30\;\text{mm} \]}

\subsubsection*{Varianza muestral ($s^2$)}

\paso{1} F\'ormula:
\formula{s^2 = \frac{\displaystyle\sum_{i=1}^{n}(x_i - \bar{x})^2}{n-1}}

\paso{2} Con $\bar{x} = 199.9964$\,mm, tres t\'erminos representativos:
\begin{align*}
(199.2 - 199.9964)^2 &= (-0.7964)^2 = 0.6342 \\
(202.1 - 199.9964)^2 &= (\phantom{-}2.1036)^2 = 4.4251 \\
(197.8 - 199.9964)^2 &= (-2.1964)^2 = 4.8242 \\
&\vdots
\end{align*}

\paso{3} Suma total de cuadrados de desviaci\'on:
\formula{\sum_{i=1}^{110}(x_i - \bar{x})^2 = 146.914}

\paso{4} Dividir entre $n - 1 = 109$:
\formula{s^2 = \frac{146.914}{109} = 1.3478\;\text{mm}^2}

\resultado[$s^2$ -- Varianza]{\[ s^2 = 1.3478\;\text{mm}^2 \]}

\subsubsection*{Desviaci\'on est\'andar ($s$)}

\paso{1} F\'ormula y c\'alculo:
\formula{s = \sqrt{s^2} = \sqrt{1.3478} = 1.1610\;\text{mm}}

\resultado[$s$ -- Desviaci\'on est\'andar]{\[ s = 1.1610\;\text{mm} \]}

\subsubsection*{Coeficiente de variaci\'on ($CV$)}

\paso{1} F\'ormula y c\'alculo:
\formula{CV = \frac{s}{\bar{x}} \times 100\% = \frac{1.1610}{199.9964} \times 100\% = 0.581\%}

\resultado[$CV$ -- Coeficiente de variaci\'on]{\[ CV = 0.581\% \]}

\interpretacion[Dispersi\'on --- Problema 2]{
El $CV = 0.581\% \ll 10\%$ indica una variabilidad \textbf{extremadamente baja
y aceptable}. La desviaci\'on est\'andar de 1.16\,mm representa solo el 19\%
de la franja de tolerancia total (6\,mm). La variabilidad del proceso es
\textbf{aceptable}.}

\interpretacion[Histograma --- Problema 2]{
La distribuci\'on presenta un patr\'on \textbf{bimodal} con un primer pico en
[198.3, 199.4)\,mm y un segundo pico mayor en [200.5, 201.6)\,mm.
Esto sugiere que las \textbf{dos m\'aquinas} tienen medias diferentes.
Todos los datos est\'an dentro de [197, 203]\,mm (proceso \textbf{capaz}), pero
se recomienda \textbf{analizar cada m\'aquina por separado} y ajustar ambas
hacia el nominal de 200\,mm.}
